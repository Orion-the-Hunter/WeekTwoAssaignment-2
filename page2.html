In JavaScript, var, let, and const are used to declare variables, but they have different scopes and behaviors. 'var' is function-scoped means it is accessible within the function. It can be updated. It is moostly used older code. Both 'let', 'const' is block-scoped. however, a const variable cannot be updated or re-declared like 'let'. It's generally recommended to use 'let' and 'constt' in modern JavaScript. function example() { var varVariable = "var"; // Accessible within this function if (true) { let letVariable = "let"; // Accessible only within this block const constVariable = "const"; // Accessible only within this block console.log(varVariable); // Outputs: "var" console.log(letVariable); // Outputs: "let" console.log(constVariable); // Outputs: "const" } console.log(varVariable); // Outputs: "var" console.log(letVariable); // ReferenceError console.log(constVariable); // ReferenceError }
